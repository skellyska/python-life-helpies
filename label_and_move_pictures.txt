import os
import shutil

# Set the directory where your pictures are located
src_dir = '/Users/Desktop/Pictures'

# Set the directory where you want to move the labeled pictures
dst_dir = '/Users/Desktop/LabeledPictures'

# Loop through all files in the source directory
for file_name in os.listdir(src_dir):
  # Check if the file is a picture (assumes all picture files end with '.jpg')
  if file_name.endswith('.jpg'):
    # Label the picture with the file name (without the '.jpg' extension)
    label = file_name[:-4]
    print(f'Labeling picture "{file_name}" with "{label}"')
    
    # Construct the full file paths
    src_path = os.path.join(src_dir, file_name)
    dst_path = os.path.join(dst_dir, file_name)
    
    # Move the picture to the labeled pictures directory
    shutil.move(src_path, dst_path)
